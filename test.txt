pipeline {
  agent any
  stages {
    stage('Download sources') {
      steps {
        sh '''
        sudo rm -rf kickstart-docker
        sudo rm -rf docker-images-build
        sudo git clone https://github.com/sloopstash/kickstart-docker.git kickstart-docker
        sudo git clone https://github.com/Eliyaser/docker-images-build.git kickstart-docker
        '''
      }
    }
    stage('Build and Push') {
     sh '''
      cd /opt/docker-images-build
      sudo chmod +x script.sh
      ./script.sh'''
    }

  }
}
